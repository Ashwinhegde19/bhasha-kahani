// Preload next 2 audio segments
const preloadNextAudio = (currentNodeId: string) => {
  const nextNodes = getNextNodes(currentNodeId);
  nextNodes.slice(0, 2).forEach(node => {
    const audio = new Audio();
    audio.preload = 'auto';
    audio.src = node.audioUrl;
  });
};
